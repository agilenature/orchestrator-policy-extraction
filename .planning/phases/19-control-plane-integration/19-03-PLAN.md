## Task 1: ConstraintBriefing model + generate_briefing()

Create `src/pipeline/live/governor/briefing.py`:

```python
from __future__ import annotations
from pydantic import BaseModel
from typing import Any

SEVERITY_ORDER = {"forbidden": 0, "requires_approval": 1, "warning": 2}

class ConstraintBriefing(BaseModel, frozen=True):
    """Grouped, severity-ordered constraint briefing for a session."""
    by_axis: dict[str, list[dict[str, Any]]] = {}  # ccd_axis -> [constraint dicts]
    total_count: int = 0
    top_severity: str | None = None  # "forbidden" | "requires_approval" | "warning" | None

def generate_briefing(constraints: list[dict[str, Any]]) -> ConstraintBriefing:
    """Group constraints by ccd_axis, order each group by severity."""
    if not constraints:
        return ConstraintBriefing()

    by_axis: dict[str, list[dict[str, Any]]] = {}
    for c in constraints:
        axis = c.get("ccd_axis") or "uncategorized"
        by_axis.setdefault(axis, []).append(c)

    # Sort each axis group by severity
    for axis in by_axis:
        by_axis[axis].sort(
            key=lambda c: SEVERITY_ORDER.get(c.get("severity", "warning"), 99)
        )

    # Determine top severity across all constraints
    severities = [c.get("severity", "warning") for c in constraints]
    top = min(severities, key=lambda s: SEVERITY_ORDER.get(s, 99)) if severities else None

    return ConstraintBriefing(
        by_axis=by_axis,
        total_count=len(constraints),
        top_severity=top,
    )
```

## Task 2: GovernorDaemon

Create `src/pipeline/live/governor/daemon.py`:

```python
from __future__ import annotations
import duckdb
from typing import Any
from .briefing import ConstraintBriefing, generate_briefing

class GovernorDaemon:
    """Reads active constraints from DuckDB and generates constraint briefings.

    Stateless between requests — reads fresh on each call.
    Opens DuckDB read-only to respect single-writer invariant.
    """

    def __init__(self, db_path: str = "data/ope.db") -> None:
        self._db_path = db_path

    def get_briefing(self, session_id: str, run_id: str) -> ConstraintBriefing:
        """Return a constraint briefing for the given session.

        Reads all active constraints from data/constraints.json via DuckDB
        or falls back to reading constraints.json directly.
        The constraints.json is the source of truth (ConstraintStore).
        """
        constraints = self._load_active_constraints()
        return generate_briefing(constraints)

    def _load_active_constraints(self) -> list[dict[str, Any]]:
        """Load active constraints from the constraint store."""
        try:
            import json
            from pathlib import Path
            path = Path("data/constraints.json")
            if not path.exists():
                return []
            data = json.loads(path.read_text())
            constraints = data if isinstance(data, list) else data.get("constraints", [])
            # Filter to active (not retired/superseded)
            return [
                c for c in constraints
                if c.get("status", "active") not in ("retired", "superseded")
            ]
        except Exception:
            return []

    def to_wire_format(self, briefing: ConstraintBriefing) -> list[dict[str, Any]]:
        """Flatten briefing to a list suitable for CheckResponse.constraints."""
        result = []
        for axis_constraints in briefing.by_axis.values():
            result.extend(axis_constraints)
        return result
```

def create_app(db_path: str = "data/ope.db", daemon=None) -> Starlette:
    ...
    from ..governor.daemon import GovernorDaemon
    _daemon = daemon or GovernorDaemon(db_path)

    async def check(request: Request) -> JSONResponse:
        try:
            body = await request.json()
            session_id = body.get("session_id", "")
            run_id = body.get("run_id", "")
            briefing = _daemon.get_briefing(session_id, run_id)
            constraints = _daemon.to_wire_format(briefing)
            return JSONResponse(CheckResponse(constraints=constraints).model_dump())
        except Exception:
            return JSONResponse(CheckResponse().model_dump())
    ...
