### `src/pipeline/review/collector.py`
Interactive verdict + opinion collection. Accepts input from stdin; mockable for tests.

```python
class VerdictCollector:
    def __init__(self, input_fn=input):
        self._input = input_fn  # injectable for testing

    def collect(self) -> tuple[ReviewVerdict, Optional[str]]:
        """
        Prompts for verdict and optional opinion.
        Loops until valid verdict entered.
        Returns (verdict, opinion_or_None).
        opinion is required to be non-empty only as a reminder;
        it is not enforced here (router handles the routing logic).
        """
        while True:
            raw = self._input("Your verdict [accept/reject]: ").strip().lower()
            if raw in ("accept", "a"):
                verdict = ReviewVerdict.ACCEPT
                break
            elif raw in ("reject", "r"):
                verdict = ReviewVerdict.REJECT
                break
            else:
                print("Please enter 'accept' or 'reject'.")
        opinion = self._input("Your opinion (optional, press Enter to skip): ").strip()
        return verdict, opinion or None
```

### `src/pipeline/review/writer.py`
Append-only writer. Never issues UPDATE or DELETE.

```python
class ReviewWriter:
    def __init__(self, conn):
        self._conn = conn

    def write(self, review: IdentificationReview) -> None:
        """
        Inserts one row to identification_reviews.
        Raises ReviewWriterError if review_id or identification_instance_id already exists.
        Never issues UPDATE or DELETE — append-only contract.
        """
        try:
            self._conn.execute(
                """
                INSERT INTO identification_reviews (
                    review_id, identification_instance_id, layer, point_id,
                    pipeline_component, trigger_text, observation_state, action_taken,
                    downstream_impact, provenance_pointer, verdict, opinion,
                    reviewed_at, session_id
                ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
                """,
                [
                    review.review_id,
                    review.identification_instance_id,
                    review.layer.value,
                    review.point_id,
                    review.pipeline_component,
                    review.trigger,
                    review.observation_state,
                    review.action_taken,
                    review.downstream_impact,
                    review.provenance_pointer,
                    review.verdict.value,
                    review.opinion,
                    review.reviewed_at,
                    review.session_id,
                ],
            )
        except duckdb.IntegrityError as e:
            raise ReviewWriterError(f"Append-only violation: {e}") from e

class ReviewWriterError(Exception):
    pass
```

```python
from src.pipeline.review.pool_builder import PoolBuilder
from src.pipeline.review.sampler import BalancedLayerSampler
from src.pipeline.review.presenter import present
from src.pipeline.review.collector import VerdictCollector
from src.pipeline.review.writer import ReviewWriter
from src.pipeline.review.models import IdentificationReview, ReviewVerdict
from datetime import datetime, timezone

@click.group()
def review():
    """Identification transparency review commands."""
    pass

@review.command("next")
@click.option("--db", default="data/ope.db", help="DuckDB path")
def review_next(db: str):
    """Present one unreviewed identification instance and collect a verdict."""
    import duckdb
    conn = duckdb.connect(db)
    _ensure_schema(conn)

    builder = PoolBuilder(conn)
    pool = builder.build()

    sampler = BalancedLayerSampler(pool, conn)
    instance = sampler.sample_one()

    if instance is None:
        click.echo("All identification instances have been reviewed.")
        return

    click.echo(present(instance))

    collector = VerdictCollector()
    verdict, opinion = collector.collect()

    review = IdentificationReview(
        identification_instance_id=instance.instance_id,
        layer=instance.layer,
        point_id=instance.point_id,
        pipeline_component=instance.pipeline_component,
        trigger=instance.trigger,
        observation_state=instance.observation_state,
        action_taken=instance.action_taken,
        downstream_impact=instance.downstream_impact,
        provenance_pointer=instance.provenance_pointer,
        verdict=verdict,
        opinion=opinion,
        reviewed_at=datetime.now(timezone.utc).isoformat(),
    )

    writer = ReviewWriter(conn)
    writer.write(review)

    status = "✓ Accepted" if verdict == ReviewVerdict.ACCEPT else "✗ Rejected"
    click.echo(f"\n{status} — review written.")
    if verdict == ReviewVerdict.REJECT and opinion:
        click.echo(f"Opinion recorded. Spec-correction candidate will be generated (run: review route).")
    elif verdict == ReviewVerdict.REJECT and not opinion:
        click.echo("Tip: rejected verdicts with opinions generate named spec-correction targets.")
```

## Key Decisions

- `VerdictCollector.input_fn` is injectable — enables test isolation without subprocess calls
- `reviewer_session_id` is not captured in this plan (no session tracking infrastructure yet); left as None — extension point for Phase 14+ when session IDs are available in live governance
- The CLI prints a routing hint when verdict=reject but opinion is empty — encouraging the human to add one, without enforcing it (enforcement is in the router, Plan 13.3-03)
- `_ensure_schema(conn)` called at CLI entry point — idempotent schema creation covering `identification_reviews` and `layer_coverage_snapshots` tables
- `__main__.py` registration follows the existing pattern: lazy import + `_setup_logging()` call before group registration
