   ```python
   def get_briefing(
       self, session_id: str, run_id: str, repo: str | None = None
   ) -> ConstraintBriefing:
       """Return a constraint briefing for the given session.

       Reads all active constraints from constraints.json on each call
       (stateless). Filters by repo scope when repo is provided:
       - Constraints with no repo_scope (None/absent) are universal
       - Constraints with repo_scope list deliver only to matching repos

       Args:
           session_id: Requesting session identifier.
           run_id: Run identifier for the session.
           repo: Repository name for scope filtering. None = deliver all.

       Returns:
           ConstraintBriefing with severity-sorted active constraints.
       """
       constraints = self._load_active_constraints()
       if repo is not None:
           constraints = self._filter_by_repo(constraints, repo)
       return generate_briefing(constraints)
   ```

2. Add `_filter_by_repo()` method:
   ```python
   def _filter_by_repo(
       self, constraints: list[dict[str, Any]], repo: str
   ) -> list[dict[str, Any]]:
       """Filter constraints to those whose repo_scope includes the given repo.

       Constraints with no repo_scope field, or repo_scope=None, or
       repo_scope=[] are universal -- delivered to ALL sessions.
       Constraints with a non-empty repo_scope list are delivered only
       when the requesting repo appears in the list.

       Args:
           constraints: List of active constraint dicts.
           repo: Repository name to filter by.

       Returns:
           Filtered list of constraints applicable to this repo.
       """
       result = []
       for c in constraints:
           repo_scope = c.get("repo_scope")
           if repo_scope is None or not isinstance(repo_scope, list) or len(repo_scope) == 0:
               # Universal constraint: no scope restriction
               result.append(c)
           elif repo in repo_scope:
               # Scoped constraint: repo matches
               result.append(c)
           # else: scoped constraint but repo does not match -- skip
       return result
   ```
