The module has four responsibilities: (1) build EdgeRecord for an accepted genus, (2) build FlameEvent for genus_shift, (3) append staging records to `data/genus_staging.jsonl`, (4) ingest staging into ope.db.

```python
"""GenusEdgeWriter for Genus-Check Gate (Phase 24).

Writes accepted genus declarations as:
- EdgeRecord to axis_edges table (relationship_text='genus_of', abstraction_level=3)
- FlameEvent to flame_events table (marker_type='genus_shift', subject='ai')

Both writes use the staging pattern (data/genus_staging.jsonl) to avoid
DuckDB single-writer conflicts with the batch pipeline. The PAG hook appends
to genus_staging.jsonl; runner.py's Step 11.6 calls ingest_genus_staging().

Exports:
    GenusEdgeWriter
    append_genus_staging
    ingest_genus_staging
"""

from __future__ import annotations

import fcntl
import json
import logging
from datetime import datetime, timezone
from pathlib import Path
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    import duckdb

from src.pipeline.ddf.models import FlameEvent
from src.pipeline.ddf.topology.models import ActivationCondition, EdgeRecord

logger = logging.getLogger(__name__)

GENUS_STAGING_PATH = "data/genus_staging.jsonl"


class GenusEdgeWriter:
    """Builds EdgeRecord and FlameEvent objects for accepted genus declarations.

    Stateless — construct once and reuse across multiple genus declarations
    in the same hook invocation.
    """

    def build_genus_edge(
        self,
        genus_name: str,
        premise_claim: str,
        session_id: str,
        instances: list[str],
    ) -> EdgeRecord:
        """Build an EdgeRecord representing a genus-of relationship.

        The edge connects the genus_name axis to the premise's claim axis,
        with relationship_text='genus_of' and abstraction_level=3.

        Args:
            genus_name: The accepted genus name (GENUS: field value).
            premise_claim: The PREMISE claim text (axis_b in the edge).
            session_id: Current session ID.
            instances: Citable instances from INSTANCES: sub-field.

        Returns:
            EdgeRecord ready to write to axis_edges.
        """
        edge_id = EdgeRecord.make_id(
            axis_a=genus_name,
            axis_b=premise_claim[:100],  # Truncate long claims
            relationship_text="genus_of",
        )
        return EdgeRecord(
            edge_id=edge_id,
            axis_a=genus_name,
            axis_b=premise_claim[:100],
            relationship_text="genus_of",
            activation_condition=ActivationCondition(
                goal_type=["write_class"],
                scope_prefix="",
                min_axes_simultaneously_active=1,
            ),
            evidence={
                "instances": instances,
                "source": "genus_check_gate",
                "session_id": session_id,
            },
            abstraction_level=3,
            status="candidate",
            trunk_quality=1.0,
            created_session_id=session_id,
        )

    def build_genus_shift_event(
        self,
        genus_name: str,
        session_id: str,
        evidence_excerpt: str | None = None,
    ) -> FlameEvent:
        """Build a FlameEvent for a genus_shift signal.

        Emitted when a valid genus is declared — signals that the AI has
        identified a new genus-level category for the problem being solved.

        Args:
            genus_name: The accepted genus name.
            session_id: Current session ID.
            evidence_excerpt: Optional excerpt from the PREMISE block.

        Returns:
            FlameEvent with marker_type='genus_shift', subject='ai'.
        """
        flame_id = FlameEvent.make_id(
            session_id=session_id,
            prompt_number=None,
            marker_type=f"genus_shift:{genus_name[:40]}",
        )
        return FlameEvent(
            flame_event_id=flame_id,
            session_id=session_id,
            marker_level=2,  # Level 2: axis identification / genus declaration
            marker_type="genus_shift",
            evidence_excerpt=evidence_excerpt or genus_name[:200],
            axis_identified=genus_name,
            subject="ai",
            detection_source="stub",
            deposited_to_candidates=False,
        )


def append_genus_staging(
    records: list[dict],
    staging_path: str = GENUS_STAGING_PATH,
) -> int:
    """Append genus staging records to genus_staging.jsonl.

    Uses file locking (fcntl.flock) for concurrent hook invocations.
    Separate from premise_staging.jsonl to avoid schema conflicts.

    Each record dict contains:
        - edge: serialized EdgeRecord dict
        - flame_event: serialized FlameEvent dict
        - session_id: str
        - created_at: ISO 8601 str

    Args:
        records: List of genus staging record dicts.
        staging_path: Path to the staging JSONL file.

    Returns:
        Number of records written.
    """
    if not records:
        return 0

    path = Path(staging_path)
    path.parent.mkdir(parents=True, exist_ok=True)

    written = 0
    with path.open("a", encoding="utf-8") as f:
        fcntl.flock(f.fileno(), fcntl.LOCK_EX)
        try:
            for record in records:
                f.write(json.dumps(record) + "\n")
                written += 1
        finally:
            fcntl.flock(f.fileno(), fcntl.LOCK_UN)

    return written


def ingest_genus_staging(
    conn: "duckdb.DuckDBPyConnection",
    staging_path: str = GENUS_STAGING_PATH,
) -> dict[str, int]:
    """Ingest genus_staging.jsonl into axis_edges and flame_events.

    Reads all records from genus_staging.jsonl, writes edge records to
    axis_edges and flame events to flame_events, then clears the staging
    file (only after all writes succeed).

    The conn must be a read-write DuckDB connection. Never called from
    the PAG hook — only called from runner.py Step 11.6.

    Args:
        conn: Read-write DuckDB connection with axis_edges and flame_events tables.
        staging_path: Path to the staging JSONL file.

    Returns:
        Dict with keys: 'edges_written', 'events_written', 'errors'.
    """
    path = Path(staging_path)
    if not path.exists():
        return {"edges_written": 0, "events_written": 0, "errors": 0}

    lines = path.read_text(encoding="utf-8").splitlines()
    if not lines:
        return {"edges_written": 0, "events_written": 0, "errors": 0}

    from src.pipeline.ddf.topology.writer import EdgeWriter
    from src.pipeline.ddf.writer import write_flame_events

    edge_writer = EdgeWriter(conn)
    edges_written = 0
    events_written = 0
    errors = 0

    edges_to_write: list[EdgeRecord] = []
    events_to_write: list[FlameEvent] = []

    for line in lines:
        line = line.strip()
        if not line:
            continue
        try:
            record = json.loads(line)
            # Reconstruct EdgeRecord
            edge_data = record.get("edge")
            if edge_data:
                ac = ActivationCondition(**edge_data.pop("activation_condition", {}))
                # Remove non-EdgeRecord keys before construction
                edge_data.pop("created_at", None)
                edge = EdgeRecord(activation_condition=ac, **edge_data)
                edges_to_write.append(edge)
            # Reconstruct FlameEvent
            fe_data = record.get("flame_event")
            if fe_data:
                fe_data.pop("created_at", None)
                event = FlameEvent(**fe_data)
                events_to_write.append(event)
        except Exception as e:
            logger.warning("Genus staging parse error: {}", e)
            errors += 1

    # Write edges (idempotent via INSERT OR REPLACE)
    for edge in edges_to_write:
        try:
            edge_writer.write_edge(edge)
            edges_written += 1
        except Exception as e:
            logger.warning("Genus edge write failed: {}", e)
            errors += 1

    # Write flame events (idempotent via INSERT OR REPLACE)
    if events_to_write:
        try:
            result = write_flame_events(conn, events_to_write)
            events_written = result.get("written", 0)
        except Exception as e:
            logger.warning("Genus flame_events write failed: {}", e)
            errors += len(events_to_write)

    # Clear staging file only after successful writes
    if edges_written + events_written > 0:
        path.write_text("", encoding="utf-8")

    return {"edges_written": edges_written, "events_written": events_written, "errors": errors}
```
