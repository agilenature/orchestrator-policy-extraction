**writer.py** -- `EdgeWriter` class:

```python
class EdgeWriter:
    def __init__(self, conn: duckdb.DuckDBPyConnection):
        self._conn = conn

    def write_edge(self, edge: EdgeRecord) -> dict[str, int]:
        """Write an EdgeRecord to axis_edges table.

        Validates activation_condition is not null/empty at write time.
        Uses INSERT OR REPLACE for idempotent writes.

        Args:
            edge: EdgeRecord to write.

        Returns:
            {"written": 1} on success.

        Raises:
            ValueError: If activation_condition is null or empty dict.
        """
        # activation_condition is already non-Optional on EdgeRecord,
        # but double-check the JSON is not empty
        ac_dict = edge.activation_condition.model_dump()
        if not ac_dict or all(v is None for v in ac_dict.values()):
            raise ValueError(
                "activation_condition must have at least one non-null field"
            )

        self._conn.execute(
            """INSERT OR REPLACE INTO axis_edges (
                edge_id, axis_a, axis_b, relationship_text,
                activation_condition, evidence, abstraction_level,
                status, trunk_quality, created_session_id, created_at
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)""",
            [
                edge.edge_id, edge.axis_a, edge.axis_b,
                edge.relationship_text,
                json.dumps(ac_dict),
                json.dumps(edge.evidence),
                edge.abstraction_level,
                edge.status, edge.trunk_quality,
                edge.created_session_id,
                edge.created_at.isoformat() if edge.created_at else None,
            ]
        )
        return {"written": 1}

    def degrade_edge(self, edge_id: str, amount: float) -> float:
        """Degrade trunk_quality by amount. Returns new quality.

        Does NOT auto-retire. Caller decides via retire_edge().
        """
        row = self._conn.execute(
            "SELECT trunk_quality FROM axis_edges WHERE edge_id = ?",
            [edge_id]
        ).fetchone()
        if not row:
            raise ValueError(f"Edge {edge_id} not found")
        new_quality = max(0.0, row[0] - amount)
        self._conn.execute(
            "UPDATE axis_edges SET trunk_quality = ? WHERE edge_id = ?",
            [new_quality, edge_id]
        )
        return new_quality

    def retire_edge(self, edge_id: str) -> None:
        """Set edge status to 'superseded'.

        One-directional: no recovery. Create a new edge for the same
        relationship in a new context.
        """
        self._conn.execute(
            "UPDATE axis_edges SET status = 'superseded' WHERE edge_id = ?",
            [edge_id]
        )

    def degrade_and_maybe_retire(
        self, edge_id: str, amount: float, threshold: float = 0.3
    ) -> tuple[float, bool]:
        """Degrade trunk_quality and retire if below threshold.

        Returns (new_quality, was_retired).
        """
        new_quality = self.degrade_edge(edge_id, amount)
        retired = False
        if new_quality < threshold:
            self.retire_edge(edge_id)
            retired = True
        return new_quality, retired

    def find_edges_for_axis_pair(
        self, axis_a: str, axis_b: str, status: str = "active"
    ) -> list[dict]:
        """Find edges between two axes (either direction).

        Args:
            axis_a: First axis name.
            axis_b: Second axis name.
            status: Filter by status (default 'active').

        Returns:
            List of edge dicts.
        """
        rows = self._conn.execute(
            """SELECT * FROM axis_edges
               WHERE ((axis_a = ? AND axis_b = ?) OR (axis_a = ? AND axis_b = ?))
               AND status = ?""",
            [axis_a, axis_b, axis_b, axis_a, status]
        ).fetchall()
        cols = [desc[0] for desc in self._conn.description]
        return [dict(zip(cols, row)) for row in rows]
```
