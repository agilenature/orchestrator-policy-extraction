    1. **ShadowReporter extension** -- Add `_compute_policy_error_metrics()` method following `_compute_amnesia_metrics()` pattern:

    ```python
    def _compute_policy_error_metrics(self) -> dict:
        """Compute policy error rate from policy_error_events and shadow_mode_results.

        Returns dict with policy_error_rate (float|None), policy_errors_total,
        suppressed, surfaced_and_blocked, total_recommendations_attempted.
        Uses rolling 100-session window via most recent 100 distinct session_ids.

        CRITICAL: The denominator (total_recommendations_attempted) MUST include
        suppressed recommendations. Suppressed recs are NOT in shadow_mode_results
        (they `continue` before evaluation), so the denominator is:
            COUNT(*) FROM shadow_mode_results
            + COUNT(*) FROM policy_error_events WHERE error_type='suppressed'
        Per locked formula: (suppressed + surfaced_and_blocked) / total_attempted
        """
        try:
            # Evaluated recommendations (in shadow_mode_results)
            evaluated_row = self._conn.execute(
                "SELECT COUNT(*) FROM shadow_mode_results"
            ).fetchone()
            evaluated_count = evaluated_row[0] or 0
        except Exception:
            return {
                "policy_error_rate": None,
                "policy_errors_total": 0,
                "suppressed": 0,
                "surfaced_and_blocked": 0,
                "total_recommendations_attempted": 0,
            }

        try:
            # Policy errors (rolling 100-session window)
            row = self._conn.execute("""
                SELECT
                    COUNT(*) as total_errors,
                    COUNT(CASE WHEN error_type = 'suppressed' THEN 1 END) as suppressed,
                    COUNT(CASE WHEN error_type = 'surfaced_and_blocked' THEN 1 END) as blocked
                FROM policy_error_events
                WHERE session_id IN (
                    SELECT DISTINCT session_id FROM shadow_mode_results
                    ORDER BY session_id DESC
                    LIMIT 100
                )
            """).fetchone()
        except Exception:
            row = (0, 0, 0)

        total_errors = row[0] or 0
        suppressed = row[1] or 0
        blocked = row[2] or 0

        # Denominator: evaluated + suppressed (suppressed are NOT in shadow_mode_results)
        total_attempted = evaluated_count + suppressed

        rate = total_errors / total_attempted if total_attempted > 0 else None

        return {
            "policy_error_rate": rate,
            "policy_errors_total": total_errors,
            "suppressed": suppressed,
            "surfaced_and_blocked": blocked,
            "total_recommendations_attempted": total_attempted,
        }
    ```

    2. **CLI audit policy-errors subcommand** -- Add to `src/pipeline/cli/audit.py`:

    ```python
    @audit_group.command(name="policy-errors")
    @click.option("--json", "output_json", is_flag=True, help="Output as JSON.")
    @click.option("--db", default="data/ope.db", help="DuckDB database path.")
    def audit_policy_errors(output_json: bool, db: str) -> None:
        """Show policy error events and error rate.

        Exit codes:
          0 - Clean (error rate < 5% or no data)
          1 - Runtime error
          2 - Policy errors detected (error rate >= 5%)

        Denominator includes suppressed recommendations (not in shadow_mode_results).
        """
        _setup_logging()

        try:
            conn = get_connection(db)
            create_schema(conn)

            # Query policy error events
            try:
                rows = conn.execute(
                    "SELECT error_id, session_id, episode_id, error_type, "
                    "constraint_id, recommendation_mode, detected_at "
                    "FROM policy_error_events ORDER BY detected_at"
                ).fetchall()
            except Exception:
                rows = []

            # Compute rate with correct denominator
            # total_attempted = evaluated (shadow_mode_results) + suppressed
            try:
                evaluated_row = conn.execute(
                    "SELECT COUNT(*) FROM shadow_mode_results"
                ).fetchone()
                evaluated_count = evaluated_row[0] or 0
            except Exception:
                evaluated_count = 0

            suppressed_count = sum(1 for r in rows if r[3] == "suppressed")
            total_attempted = evaluated_count + suppressed_count

            conn.close()

            error_count = len(rows)
            rate = error_count / total_attempted if total_attempted > 0 else None

            if output_json:
                result = {
                    "policy_errors": [
                        {
                            "error_id": r[0], "session_id": r[1],
                            "episode_id": r[2], "error_type": r[3],
                            "constraint_id": r[4], "recommendation_mode": r[5],
                            "detected_at": str(r[6]) if r[6] else None,
                        }
                        for r in rows
                    ],
                    "error_count": error_count,
                    "total_attempted": total_attempted,
                    "error_rate": rate,
                }
                click.echo(json.dumps(result, indent=2))
            else:
                click.echo("Policy Error Report")
                click.echo("===================")
                click.echo(f"Total policy errors: {error_count}")
                click.echo(f"Total recommendations: {total_attempted}")
                if rate is not None:
                    gate = "PASS" if rate < 0.05 else "FAIL"
                    click.echo(f"Error rate: {rate:.1%} (target: <5%)  {gate}")
                else:
                    click.echo("Error rate: N/A (no recommendations)")

                if rows:
                    click.echo("")
                    blocked_count = sum(1 for r in rows if r[3] == "surfaced_and_blocked")
                    click.echo(f"  Suppressed: {suppressed_count}")
                    click.echo(f"  Surfaced & blocked: {blocked_count}")

            if rate is not None and rate >= 0.05:
                sys.exit(2)
            sys.exit(0)

        except SystemExit:
            raise
        except Exception as e:
            click.echo(f"Error: {e}", err=True)
            sys.exit(1)
    ```
