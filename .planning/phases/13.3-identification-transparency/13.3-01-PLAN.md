### `src/pipeline/review/models.py`
```python
from enum import Enum
from pydantic import BaseModel, Field
from typing import Optional
import uuid

class IdentificationLayer(str, Enum):
    L1_EVENT_FILTER = "L1"        # Event filtering and actor assignment
    L2_TAGGING = "L2"             # Tagging
    L3_SEGMENTATION = "L3"        # Segmentation
    L4_EPISODE_POPULATION = "L4"  # Episode population
    L5_CONSTRAINT_EXTRACTION = "L5"
    L6_CONSTRAINT_EVALUATION = "L6"
    L7_ESCALATION_DETECTION = "L7"
    L8_POLICY_FEEDBACK = "L8"

class ReviewVerdict(str, Enum):
    ACCEPT = "accept"
    REJECT = "reject"

class IdentificationPoint(BaseModel):
    """One classification act the pipeline has performed."""
    instance_id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    layer: IdentificationLayer
    point_id: str                  # e.g. "L2-1", "L4-4"
    point_label: str               # human-readable name
    pipeline_component: str        # e.g. "OrchestratorTagger", "ReactionLabeler"
    # Five externalization properties
    trigger: str                   # what prompted this classification
    observation_state: str         # the raw input the classifier saw
    action_taken: str              # the label/classification chosen
    downstream_impact: str         # what decisions depended on this
    provenance_pointer: str        # session_id + event_id + episode_id + source_file:line
    # Source artifact references
    source_session_id: Optional[str] = None
    source_event_id: Optional[str] = None
    source_episode_id: Optional[str] = None

class IdentificationReview(BaseModel):
    """One completed review of an identification instance."""
    review_id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    identification_instance_id: str
    layer: IdentificationLayer
    point_id: str
    pipeline_component: str
    # Five externalization properties (copied from instance for self-contained storage)
    trigger: str
    observation_state: str
    action_taken: str
    downstream_impact: str
    provenance_pointer: str
    # Verdict
    verdict: ReviewVerdict
    opinion: Optional[str] = None   # required path when verdict=reject
    reviewed_at: str                 # ISO-8601 UTC
    session_id: Optional[str] = None # the session in which this review was performed
```

### `src/pipeline/review/pool_builder.py`
Sources one IdentificationPoint per pipeline artifact for each of the 35 identification point types.

Key design decisions:
- Reads from existing DuckDB tables (events, episodes, constraints, escalation_events, session_constraint_eval, amnesia_events, policy_error_events)
- Returns at most N instances per point_id to prevent any single source dominating
- Each instance carries a traceable provenance_pointer with session_id + event/episode ID + source table + row key
- Falls back gracefully when a source table has no rows (returns empty list for that point_id)

```python
class PoolBuilder:
    def __init__(self, conn, max_per_point: int = 10):
        self.conn = conn
        self.max_per_point = max_per_point

    def build(self) -> list[IdentificationPoint]:
        """Return pool covering all 35 point_ids, up to max_per_point each."""
        instances = []
        instances.extend(self._build_l1())
        instances.extend(self._build_l2())
        instances.extend(self._build_l3())
        instances.extend(self._build_l4())
        instances.extend(self._build_l5())
        instances.extend(self._build_l6())
        instances.extend(self._build_l7())
        instances.extend(self._build_l8())
        return instances

    # Private builders per layer — each returns list[IdentificationPoint]
    # L1: read from events table
    # L2: read from events table (tagged events with confidence, secondary_labels, mode, risk)
    # L3: read from episode_segments table (start_trigger, end_trigger, outcome, complexity)
    # L4: read from episodes table (observation, action, outcome, reaction_label, confidence, mode, risk)
    # L5: read from constraints table (linked to episode via examples)
    # L6: read from session_constraint_eval table (honored/violated/evidence)
    # L7: read from escalation_events table (block/bypass/sequence/constraint_bypassed)
    # L8: read from policy_error_events table (suppressed/surfaced_blocked/duplicate)
```

### `src/pipeline/review/sampler.py`
Balanced layer sampler ensuring no layer > 20% of output when N >= 40.

```python
class BalancedLayerSampler:
    """
    Samples uniformly across 8 layers.
    Excludes already-reviewed instances (those with a verdict in identification_reviews).
    """
    def __init__(self, pool: list[IdentificationPoint], conn):
        self.pool = pool
        self.conn = conn

    def sample_one(self) -> Optional[IdentificationPoint]:
        """
        Returns one unreviewed instance, selected from the layer with the
        lowest current coverage ratio (reviewed/available).
        Returns None if all instances have been reviewed.
        """
        reviewed_ids = self._get_reviewed_ids()
        unreviewed = [p for p in self.pool if p.instance_id not in reviewed_ids]
        if not unreviewed:
            return None
        # Select from the layer with lowest coverage
        layer_coverage = self._compute_layer_coverage(reviewed_ids)
        target_layer = min(layer_coverage, key=lambda l: layer_coverage[l])
        candidates = [p for p in unreviewed if p.layer == target_layer]
        if not candidates:
            # Fallback: any unreviewed
            candidates = unreviewed
        return random.choice(candidates)
```

## Key Decisions

- `identification_reviews` UNIQUE constraint on `identification_instance_id` — one verdict per instance. This enforces at-most-once semantics at the DB level.
- Append-only is enforced in code: `writer.py` (Plan 13.3-02) never issues UPDATE or DELETE to `identification_reviews`; any attempt raises a ReviewWriterError
- CCD format constraint on `memory_candidates` is additive (ALTER TABLE ADD COLUMN IF NOT EXISTS) — backward-compatible with existing rows
- Pool builder uses LIMIT + ORDER BY RANDOM() in SQL to get varied samples without loading entire tables
- `source_instance_id` on `memory_candidates` creates the link from spec-correction candidates back to the identification review that triggered them (closes the specification_closure harness invariant)
