@intelligence_group.group(name="edges")

@edges_group.command(name="list")
@click.option("--db", default="data/ope.db", help="DuckDB database path.")
@click.option("--axis", default=None, help="Filter edges involving this axis.")
def list_edges(db: str, axis: str | None) -> None:
    """List active axis edges with relationship and activation_condition.

    Usage:
        python -m src.pipeline.cli intelligence edges list
        python -m src.pipeline.cli intelligence edges list --axis deposit-not-detect
    """
    _setup_logging()

    try:
        import duckdb

        conn = duckdb.connect(db, read_only=True)

        if axis:
            rows = conn.execute(
                """SELECT edge_id, axis_a, axis_b, relationship_text,
                          activation_condition, trunk_quality
                   FROM axis_edges
                   WHERE status = 'active' AND (axis_a = ? OR axis_b = ?)
                   ORDER BY trunk_quality DESC""",
                [axis, axis],
            ).fetchall()
        else:
            rows = conn.execute(
                """SELECT edge_id, axis_a, axis_b, relationship_text,
                          activation_condition, trunk_quality
                   FROM axis_edges
                   WHERE status = 'active'
                   ORDER BY trunk_quality DESC"""
            ).fetchall()

        conn.close()

        if not rows:
            click.echo("No active edges found.")
            sys.exit(0)

        for edge_id, ax_a, ax_b, rel, ac, tq in rows:
            click.echo(f"\n  Edge: {edge_id}")
            click.echo(f"  {ax_a} <-> {ax_b}")
            rel_display = rel[:80] + "..." if len(rel) > 80 else rel
            click.echo(f"  Relationship: {rel_display}")
            click.echo(f"  Activation: {ac}")
            click.echo(f"  Trunk Quality: {tq:.2f}")

        click.echo(f"\nTotal active edges: {len(rows)}")

    except SystemExit:
        raise
    except Exception as e:
        click.echo(f"Error: {e}", err=True)
        sys.exit(1)


@edges_group.command(name="frontier")
@click.option("--db", default="data/ope.db", help="DuckDB database path.")
def frontier(db: str) -> None:
    """Show axis pairs active in recent sessions with no recorded edge.

    Identifies 'blank spaces' on the AI's Causal Map -- pairs of CCD
    axes that have been simultaneously active but have no relationship
    recorded in axis_edges.

    Usage:
        python -m src.pipeline.cli intelligence edges frontier
    """
    _setup_logging()

    try:
        import duckdb

        conn = duckdb.connect(db, read_only=True)

        # Find axis pairs that co-occur in flame_events but have no edge
        # Step 1: Find all distinct axes from flame_events
        try:
            axes_rows = conn.execute(
                """SELECT DISTINCT axis_identified FROM flame_events
                   WHERE axis_identified IS NOT NULL
                   AND axis_identified != ''"""
            ).fetchall()
        except Exception:
            axes_rows = []

        # Also include axes from memory_candidates
        try:
            mc_rows = conn.execute(
                "SELECT DISTINCT ccd_axis FROM memory_candidates"
            ).fetchall()
            axes_rows = axes_rows + mc_rows
        except Exception:
            pass

        known_axes = sorted(set(row[0] for row in axes_rows if row[0]))

        if len(known_axes) < 2:
            click.echo("Fewer than 2 known axes -- no frontier pairs possible.")
            conn.close()
            sys.exit(0)

        from src.pipeline.ddf.topology.frontier import FrontierChecker

        checker = FrontierChecker(conn)
        frontier_pairs = checker.find_frontier_pairs(known_axes)
        conn.close()

        if not frontier_pairs:
            click.echo("No frontier pairs found -- all axis pairs have active edges.")
            sys.exit(0)

        click.echo("Frontier Pairs (no active edge):")
        click.echo("=" * 50)
        for ax_a, ax_b in frontier_pairs:
            click.echo(f"  {ax_a} <-> {ax_b}")

        click.echo(f"\nTotal frontier pairs: {len(frontier_pairs)}")

    except SystemExit:
        raise
    except Exception as e:
        click.echo(f"Error: {e}", err=True)
        sys.exit(1)


@edges_group.command(name="show")
@click.argument("edge_id")
@click.option("--db", default="data/ope.db", help="DuckDB database path.")
def show_edge(edge_id: str, db: str) -> None:
    """Show full edge artifact with evidence, trunk_quality, and status.

    Usage:
        python -m src.pipeline.cli intelligence edges show EDGE_ID
    """
    _setup_logging()

    try:
        import json
        import duckdb

        conn = duckdb.connect(db, read_only=True)
        row = conn.execute(
            "SELECT * FROM axis_edges WHERE edge_id = ?",
            [edge_id],
        ).fetchone()
        conn.close()

        if not row:
            click.echo(f"Edge not found: {edge_id}")
            sys.exit(1)

        cols = [
            "edge_id", "axis_a", "axis_b", "relationship_text",
            "activation_condition", "evidence", "abstraction_level",
            "status", "trunk_quality", "created_session_id", "created_at",
        ]
        edge = dict(zip(cols, row))

        click.echo(f"Edge: {edge['edge_id']}")
        click.echo("=" * 50)
        click.echo(f"  Axis A: {edge['axis_a']}")
        click.echo(f"  Axis B: {edge['axis_b']}")
        click.echo(f"  Status: {edge['status']}")
        click.echo(f"  Trunk Quality: {edge['trunk_quality']:.2f}")
        click.echo(f"  Abstraction Level: {edge['abstraction_level']}")
        click.echo(f"  Session: {edge['created_session_id']}")
        click.echo(f"  Created: {edge['created_at']}")
        click.echo(f"\n  Relationship:")
        click.echo(f"    {edge['relationship_text']}")

        # Parse and display activation_condition
        try:
            ac = json.loads(edge["activation_condition"]) if isinstance(edge["activation_condition"], str) else edge["activation_condition"]
            click.echo(f"\n  Activation Condition:")
            click.echo(f"    Goal Types: {ac.get('goal_type', [])}")
            click.echo(f"    Scope Prefix: {ac.get('scope_prefix', '')!r}")
            click.echo(f"    Min Axes Active: {ac.get('min_axes_simultaneously_active', 0)}")
        except Exception:
            click.echo(f"\n  Activation Condition: {edge['activation_condition']}")

        # Parse and display evidence
        try:
            ev = json.loads(edge["evidence"]) if isinstance(edge["evidence"], str) else edge["evidence"]
            click.echo(f"\n  Evidence:")
            click.echo(f"    Session: {ev.get('session_id', 'unknown')}")
            click.echo(f"    Episode: {ev.get('episode_id', 'unknown')}")
            fids = ev.get("flame_event_ids", [])
            click.echo(f"    Flame Events: {', '.join(fids) if fids else 'none'}")
        except Exception:
            click.echo(f"\n  Evidence: {edge['evidence']}")

    except SystemExit:
        raise
    except Exception as e:
        click.echo(f"Error: {e}", err=True)
        sys.exit(1)
