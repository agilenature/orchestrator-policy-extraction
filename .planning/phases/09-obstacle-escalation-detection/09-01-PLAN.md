```python
class EscalationCandidate(BaseModel, frozen=True):
    session_id: str
    block_event_id: str        # The O_GATE/O_CORR event ID
    block_event_tag: str       # "O_GATE" or "O_CORR"
    bypass_event_id: str       # The bypass tool call event ID
    bypass_tool_name: str      # Tool name (Bash, Edit, Write, etc.)
    bypass_command: str = ""   # Command text for constraint template
    bypass_resource: str = ""  # Resource path for scope inference
    window_turns_used: int     # How many non-exempt events between block and bypass
    confidence: float          # Detection confidence (1.0 for event-based)
    detector_version: str      # e.g. "1.0.0"
```
