```python
async def push_link(request: Request) -> JSONResponse:
    """Accept and persist a causal push link to the push_links table.

    Required fields: parent_decision_id, child_decision_id,
    transition_trigger, migration_run_id.
    Optional: link_id (auto-generated if absent), repo_boundary,
    captured_at (defaults to server time).

    Fails open: DuckDB write errors return 200 with warning.
    Returns 400 only for missing required fields.
    """
    body: dict = {}
    try:
        body = await request.json()
    except Exception:
        return JSONResponse(
            {"status": "error", "detail": "invalid JSON body"},
            status_code=400,
        )

    # Validate required fields
    required = ["parent_decision_id", "child_decision_id",
                "transition_trigger", "migration_run_id"]
    missing = [f for f in required if not body.get(f)]
    if missing:
        return JSONResponse(
            {"status": "error",
             "detail": f"missing required fields: {', '.join(missing)}"},
            status_code=400,
        )

    parent_id = body["parent_decision_id"]
    child_id = body["child_decision_id"]
    trigger = body["transition_trigger"]
    run_id = body["migration_run_id"]
    repo_boundary = body.get("repo_boundary", None)

    # Generate link_id if not provided (deterministic from content)
    link_id = body.get("link_id", "")
    if not link_id:
        raw = f"link:{parent_id}:{child_id}:{trigger}"
        link_id = hashlib.sha256(raw.encode()).hexdigest()[:16]

    captured_at = body.get(
        "captured_at",
        datetime.now(timezone.utc).isoformat(),
    )

    try:
        conn.execute(
            "INSERT OR REPLACE INTO push_links "
            "(link_id, parent_decision_id, child_decision_id, "
            "transition_trigger, repo_boundary, migration_run_id, "
            "captured_at) VALUES (?, ?, ?, ?, ?, ?, ?)",
            [link_id, parent_id, child_id, trigger,
             repo_boundary, run_id, captured_at],
        )
    except Exception:
        # Fail open: DuckDB error does not block the caller
        return JSONResponse({
            "status": "accepted",
            "link_id": link_id,
            "warning": "push link accepted but DuckDB write failed",
        })

    return JSONResponse({
        "status": "accepted",
        "link_id": link_id,
    })
```

Note: `conn` is the DuckDB connection from the `create_app` closure (same pattern as register/deregister handlers). `hashlib` needs to be imported at the top of server.py.
