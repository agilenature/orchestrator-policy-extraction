**frontier.py** -- `FrontierChecker` class:

```python
"""Frontier Warning detector for uncharted axis pairs.

Queries axis_edges to detect when two simultaneously active CCD axes
have no recorded active edge -- indicating "Frontier territory" where
the AI is operating between established principles with no recorded
relationship.

Frontier Warnings are permanent infrastructure, not scaffolding. The
PAG gate will always lack proprioception for novel axis combinations.

Exports:
    FrontierChecker
"""
from __future__ import annotations

import json
from itertools import combinations

import duckdb


class FrontierChecker:
    """Checks for uncharted axis pairs in the current context.

    Queries axis_edges WHERE status='active' and activation_condition
    matches the current goal_type and scope_prefix. For each pair of
    active axes with no matching edge, emits a Frontier Warning.
    """

    def __init__(self, conn: duckdb.DuckDBPyConnection):
        """Initialize with a READ-ONLY DuckDB connection.

        Args:
            conn: DuckDB connection (must be read-only for PAG gate safety).
        """
        self._conn = conn

    def check_frontier(
        self,
        active_axes: list[str],
        goal_type: str | None = None,
        scope_prefix: str = "",
    ) -> list[str]:
        """Check for axis pairs with no active edge.

        For each combination of 2 active axes, queries axis_edges for
        an active edge matching the activation_condition. Returns
        warning strings for uncharted pairs.

        Args:
            active_axes: CCD axes active in the current context (from
                PREMISE declarations or active_constraints).
            goal_type: Current goal type for activation_condition filtering.
            scope_prefix: Current scope prefix for activation_condition filtering.

        Returns:
            List of FRONTIER_WARNING strings for uncharted axis pairs.
        """
        warnings: list[str] = []
        unique_axes = sorted(set(active_axes))

        if len(unique_axes) < 2:
            return warnings

        for axis_a, axis_b in combinations(unique_axes, 2):
            if not self._has_active_edge(axis_a, axis_b, goal_type, scope_prefix):
                warnings.append(
                    f"FRONTIER_WARNING: Operating between [{axis_a}] and "
                    f"[{axis_b}] with no recorded relationship -- "
                    f"Frontier territory. Geological Drill zone."
                )

        return warnings

    def _has_active_edge(
        self,
        axis_a: str,
        axis_b: str,
        goal_type: str | None,
        scope_prefix: str,
    ) -> bool:
        """Check if an active edge exists for this axis pair.

        Checks both orderings (axis_a,axis_b) and (axis_b,axis_a).
        Filters by activation_condition: goal_type must be in the
        edge's goal_type list (or edge has ["any"]), and scope_prefix
        must start with the edge's scope_prefix (or edge has "").

        Args:
            axis_a: First axis.
            axis_b: Second axis.
            goal_type: Current goal type (None matches ["any"]).
            scope_prefix: Current scope prefix ("" matches any).

        Returns:
            True if a matching active edge exists.
        """
        rows = self._conn.execute(
            """SELECT activation_condition FROM axis_edges
               WHERE ((axis_a = ? AND axis_b = ?) OR (axis_a = ? AND axis_b = ?))
               AND status = 'active'""",
            [axis_a, axis_b, axis_b, axis_a],
        ).fetchall()

        for (ac_json,) in rows:
            if self._activation_matches(ac_json, goal_type, scope_prefix):
                return True

        return False

    def _activation_matches(
        self,
        ac_json: str,
        goal_type: str | None,
        scope_prefix: str,
    ) -> bool:
        """Check if activation_condition matches current context.

        Rules:
        - edge goal_type=["any"] matches everything
        - edge goal_type contains the current goal_type -> match
        - edge scope_prefix="" matches everything
        - current scope_prefix starts with edge scope_prefix -> match

        Args:
            ac_json: JSON string of activation_condition.
            goal_type: Current goal type.
            scope_prefix: Current scope prefix.

        Returns:
            True if activation_condition matches.
        """
        try:
            ac = json.loads(ac_json) if isinstance(ac_json, str) else ac_json
        except (json.JSONDecodeError, TypeError):
            return False

        # Goal type check
        edge_goals = ac.get("goal_type", ["any"])
        if "any" not in edge_goals:
            if goal_type and goal_type not in edge_goals:
                return False

        # Scope prefix check
        edge_prefix = ac.get("scope_prefix", "")
        if edge_prefix:
            if not scope_prefix.startswith(edge_prefix):
                return False

        return True

    def find_frontier_pairs(
        self,
        active_axes: list[str],
        goal_type: str | None = None,
        scope_prefix: str = "",
    ) -> list[tuple[str, str]]:
        """Return uncharted axis pairs (without warning text).

        Useful for CLI display and programmatic access.

        Args:
            active_axes: CCD axes active in context.
            goal_type: Current goal type.
            scope_prefix: Current scope prefix.

        Returns:
            List of (axis_a, axis_b) tuples with no active edge.
        """
        pairs = []
        unique_axes = sorted(set(active_axes))
        for axis_a, axis_b in combinations(unique_axes, 2):
            if not self._has_active_edge(axis_a, axis_b, goal_type, scope_prefix):
                pairs.append((axis_a, axis_b))
        return pairs
```

**verifier.py** -- `CrossAxisVerifier` class:

```python
"""Cross-Axis Registry Verification for premise-edge consistency.

When a PREMISE declaration references two CCD axes simultaneously,
verifies the claim is consistent with recorded edge relationship_text.
Also enforces foil level-matching: foil abstraction level must be
<= premise abstraction level + 1.

Exports:
    CrossAxisVerifier
"""
from __future__ import annotations

import json
from itertools import combinations

import duckdb


class CrossAxisVerifier:
    """Verifies premise claims against recorded cross-axis edges.

    Called from the PAG gate when PREMISE blocks reference multiple
    axes. Checks whether the claim is consistent with recorded
    relationship_text in axis_edges.
    """

    def __init__(self, conn: duckdb.DuckDBPyConnection):
        """Initialize with a READ-ONLY DuckDB connection.

        Args:
            conn: DuckDB connection (must be read-only).
        """
        self._conn = conn

    def verify_premise(
        self,
        premise_axes: list[str],
        premise_claim: str,
        premise_abstraction_level: int | None = None,
        goal_type: str | None = None,
        scope_prefix: str = "",
    ) -> list[str]:
        """Verify a premise claim against recorded cross-axis edges.

        For each pair of axes in premise_axes, looks up active edges
        and checks for potential inconsistencies.

        Args:
            premise_axes: CCD axes referenced in the PREMISE.
            premise_claim: The claim text from the PREMISE block.
            premise_abstraction_level: Abstraction level of the premise
                (used for foil level-matching). None skips level check.
            goal_type: Current goal type for activation_condition filtering.
            scope_prefix: Current scope prefix.

        Returns:
            List of CROSS_AXIS_WARNING strings for inconsistencies.
        """
        warnings: list[str] = []
        unique_axes = sorted(set(premise_axes))

        if len(unique_axes) < 2:
            return warnings

        for axis_a, axis_b in combinations(unique_axes, 2):
            pair_warnings = self._check_pair(
                axis_a, axis_b, premise_claim,
                premise_abstraction_level, goal_type, scope_prefix,
            )
            warnings.extend(pair_warnings)

        return warnings

    def _check_pair(
        self,
        axis_a: str,
        axis_b: str,
        premise_claim: str,
        premise_level: int | None,
        goal_type: str | None,
        scope_prefix: str,
    ) -> list[str]:
        """Check a single axis pair against recorded edges.

        Returns warnings if:
        1. An active edge exists but the premise_claim may contradict
           the recorded relationship_text.
        2. Foil level-matching violation: edge abstraction_level >
           premise_level + 1.

        Args:
            axis_a: First axis.
            axis_b: Second axis.
            premise_claim: The PREMISE claim text.
            premise_level: Abstraction level of the premise.
            goal_type: Current goal type.
            scope_prefix: Current scope prefix.

        Returns:
            List of warning strings.
        """
        warnings: list[str] = []

        rows = self._conn.execute(
            """SELECT edge_id, relationship_text, activation_condition,
                      abstraction_level, trunk_quality
               FROM axis_edges
               WHERE ((axis_a = ? AND axis_b = ?) OR (axis_a = ? AND axis_b = ?))
               AND status = 'active'""",
            [axis_a, axis_b, axis_b, axis_a],
        ).fetchall()

        for edge_id, rel_text, ac_json, edge_level, trunk_q in rows:
            # Filter by activation_condition
            if not self._activation_matches(ac_json, goal_type, scope_prefix):
                continue

            # Foil level-matching check (CONTEXT.md Q5)
            if premise_level is not None and edge_level is not None:
                if edge_level > premise_level + 1:
                    warnings.append(
                        f"CROSS_AXIS_WARNING: Edge {edge_id} between "
                        f"[{axis_a}] and [{axis_b}] is at abstraction "
                        f"level {edge_level}, but premise is at level "
                        f"{premise_level}. Foil level mismatch "
                        f"(edge level must be <= premise level + 1). "
                        f"Possible Equivocation."
                    )

            # Note: full semantic consistency checking between
            # premise_claim and relationship_text would require NLP.
            # For Phase 16.1, we emit a contextual note when an edge
            # exists, enabling the human to verify consistency.
            # The edge text is surfaced as context, not auto-validated.

        return warnings

    def _activation_matches(
        self,
        ac_json: str,
        goal_type: str | None,
        scope_prefix: str,
    ) -> bool:
        """Check activation_condition match (same logic as FrontierChecker)."""
        try:
            ac = json.loads(ac_json) if isinstance(ac_json, str) else ac_json
        except (json.JSONDecodeError, TypeError):
            return False

        edge_goals = ac.get("goal_type", ["any"])
        if "any" not in edge_goals:
            if goal_type and goal_type not in edge_goals:
                return False

        edge_prefix = ac.get("scope_prefix", "")
        if edge_prefix:
            if not scope_prefix.startswith(edge_prefix):
                return False

        return True
```

def _check_frontier_warning(
    premises: list, session_id: str, cwd: str

    """Check for Frontier Warnings: axis pairs with no recorded edge.

    Extracts CCD axes from PREMISE SCOPE fields and checks if any
    pair of simultaneously active axes has no active edge in axis_edges.

    Uses READ-ONLY DuckDB. Wrapped in try/except (fail-open).

    Args:

        session_id: Current session ID.

    Returns:
        List of FRONTIER_WARNING strings.

    if not db_path.exists():

def _check_cross_axis(
    premises: list, session_id: str, cwd: str
) -> list[str]:
    """Check premise claims against recorded cross-axis edges.

    Uses READ-ONLY DuckDB. Wrapped in try/except (fail-open).

    Args:
        premises: List of ParsedPremise objects.
        session_id: Current session ID.
        cwd: Current working directory.

    Returns:
        List of CROSS_AXIS_WARNING strings.
    """

        finally:
            conn.close()
    except Exception as e:
        logger.debug("Cross-axis check failed (fail-open): %s", e)

    return warnings


def _extract_axes_from_premises(
    premises: list, conn: duckdb.DuckDBPyConnection
) -> list[str]:
    """Extract CCD axis names mentioned across all premises.

    Heuristic: query DISTINCT ccd_axis from memory_candidates table,
    then check if any axis name appears in any premise's scope or claim.

    Args:
        premises: List of ParsedPremise objects.
        conn: Read-only DuckDB connection.

    Returns:
        List of CCD axis names found across all premises.
    """
    try:
        axes_rows = conn.execute(
            "SELECT DISTINCT ccd_axis FROM memory_candidates"
        ).fetchall()
    except Exception:
        return []

    known_axes = [row[0] for row in axes_rows]
    if not known_axes:
        return []

    found = set()
    for premise in premises:
        text = (premise.claim + " " + (premise.scope or "")).lower()
        for axis in known_axes:
            if axis.lower() in text:
                found.add(axis)

    return list(found)


def _extract_axes_from_single_premise(
    premise, conn: duckdb.DuckDBPyConnection
) -> list[str]:
    """Extract CCD axis names from a single premise.

    Args:
        premise: A ParsedPremise object.
        conn: Read-only DuckDB connection.

    Returns:
        List of CCD axis names found in this premise.
    """
    try:
        axes_rows = conn.execute(
            "SELECT DISTINCT ccd_axis FROM memory_candidates"
        ).fetchall()
    except Exception:
        return []

    known_axes = [row[0] for row in axes_rows]
    found = set()
    text = (premise.claim + " " + (premise.scope or "")).lower()
    for axis in known_axes:
        if axis.lower() in text:
            found.add(axis)

    return list(found)
