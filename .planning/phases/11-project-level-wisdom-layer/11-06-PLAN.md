1. Replace the `check_scope` function entirely:

```python
@wisdom_group.command(name="check-scope")
@click.argument("scope_path")
@click.option("--db", default="data/ope.db", help="DuckDB database path.")
@click.option(
    "--constraints",
    default="data/constraints.json",
    help="Path to constraints JSON file.",
)
def check_scope(scope_path: str, db: str, constraints: str) -> None:
    """Validate scope decisions for a given path.

    Checks whether any active scope_decision entities have associated
    constraints that apply to SCOPE_PATH. Reports violations when
    constraints linked to scope decisions overlap the given path.

    Exit codes:
      0 = no violations found (or no scope decisions match)
      1 = at least one violation found
      2 = runtime error
    """
    _setup_logging()

    try:
        from pathlib import Path as P

        from src.pipeline.constraint_store import ConstraintStore
        from src.pipeline.utils import scopes_overlap
        from src.pipeline.wisdom.store import WisdomStore

        store = WisdomStore(P(db))
        entities = store.search_by_scope(scope_path)
        scope_decisions = [e for e in entities if e.entity_type == "scope_decision"]

        if not scope_decisions:
            click.echo(f"No scope decisions found for path: {scope_path}")
            sys.exit(0)

        # Load active constraints
        constraint_store = ConstraintStore(P(constraints))
        active_constraints = constraint_store.get_active_constraints()

        violations: list[tuple[str, str, str]] = []  # (decision_title, constraint_text, severity)

        for decision in scope_decisions:
            click.echo(f"\n[scope_decision] {decision.title}")
            click.echo(f"  {decision.description}")
            if decision.scope_paths:
                click.echo(f"  Scope: {', '.join(decision.scope_paths)}")

            # Find constraints related to this scope decision by text matching:
            # A constraint is related if its text contains keywords from the
            # scope decision title (case-insensitive substring match).
            # Additionally, the constraint scope must overlap the query path.
            title_lower = decision.title.lower()
            title_words = [w for w in title_lower.split() if len(w) > 3]

            matched_constraints = []
            for c in active_constraints:
                c_text = c.get("text", "").lower()
                c_scope_paths = c.get("scope", {}).get("paths", [])

                # Check text overlap: at least 2 significant title words in constraint text
                word_matches = sum(1 for w in title_words if w in c_text)
                if word_matches < 2:
                    continue

                # Check scope overlap with the query path
                if c_scope_paths and not scopes_overlap([scope_path], c_scope_paths):
                    continue

                matched_constraints.append(c)

            for c in matched_constraints:
                severity = c.get("severity", "warning")
                c_text = c.get("text", "")
                c_id = c.get("constraint_id", "unknown")

                if severity in ("forbidden", "requires_approval"):
                    violations.append((decision.title, c_text, severity))
                    click.echo(f"  VIOLATION: constraint {c_id} ({severity})")
                    click.echo(f"    {c_text[:120]}")
                else:
                    click.echo(f"  Related constraint {c_id} ({severity}): {c_text[:80]}")

        # Report summary and exit
        if violations:
            click.echo(f"\n{len(violations)} violation(s) found.")
            sys.exit(1)
        else:
            click.echo("\nNo violations found.")
            sys.exit(0)

    except SystemExit:
        raise  # Let sys.exit() propagate
    except Exception as e:
        click.echo(f"Error: {e}", err=True)
        sys.exit(2)
```

Key changes from the current implementation:
- Added `--constraints` option to locate the constraints JSON file
- Loads ConstraintStore and gets active constraints
- Text-based matching: finds constraints related to scope decisions by matching significant title words (>3 chars) against constraint text -- requires 2+ word overlap
- Scope overlap check: uses the shared `scopes_overlap()` from utils.py to verify constraint scope applies to query path
- Violations: constraints with severity 'forbidden' or 'requires_approval' that match scope decisions
- Exit code 0: no violations (or no matching scope decisions)
- Exit code 1: at least one violation found
- Exit code 2: runtime errors (was previously exit code 1)
- The `except SystemExit: raise` pattern ensures sys.exit() calls inside the try block propagate correctly
